1   package pages.lesson1.PageObject;
2   
3   import io.qameta.allure.Step;
4   import org.openqa.selenium.WebDriver;
5   import org.openqa.selenium.WebElement;
6   import org.openqa.selenium.support.FindAll;
7   import org.openqa.selenium.support.FindBy;
8   
9   public class LoginPage extends BasePage {
10  
11      //*********Constructor*********
12      public LoginPage(WebDriver driver) {
13          super(driver);
14      }
15  
16      //*********Page Variables*********
17      String loginURL = "https://accounts.ukr.net/login?lang=ui";
18  
19      //*********Web Elements*********
20      @FindBy(css = "button.form__submit")
21      private WebElement signInButtonElement;
22      @FindBy(id = "id-l")
23      private WebElement usernameElement;
24      @FindAll({
25              @FindBy(id = "id-p"),
26              @FindBy(id = "some invalid selector")
27      })
28      private WebElement passwordElement;
29  
30      //*********Page Methods*********
31      @Step("Open login page")
32      public LoginPage goToLoginPage() {
33          driver.get(loginURL);
34          return this;
35      }
36  
37      @Step("Login to mail with login: {0}, password: {1} ")
38      public UkrNetHomePage loginToUkrNet(String username, String password) {
39          //Enter Username(Email)
40          writeText(usernameElement, username);
41          //Enter Password
42          writeText(passwordElement, password);
43          //Click Login Button
44          return goToUkrNet();
45      }
46  
47      @Step("Go to UkrNetHomePage")
48      public UkrNetHomePage goToUkrNet() {
49          click(signInButtonElement);
50          return new UkrNetHomePage(driver);
51      }
52  }
