41      /**
42       * @param receiver -- persons who gets the mail
43       * @param title    -- email's title
44       * @param letter   -- email's content
45       */
46      @Step("Creating and sending email to: {0} with title: {1} and body: {2}")
47      public void createLetter(String receiver, String title, String letter) {
48          click(writeLetterButtonElement);
49          writeText(toElement, receiver);
50          writeText(titleElement, title);
51          writeTextToFrame(mce_0_ifr, tinymce, letter);
52          click(send);
53      }

55      @Step("Verify sending letter Popup presence")
56      public UkrNetHomePage verifySendingLetterPopup() {
57          assertTrue(sentMailPopUpElement, !readText(sentMailPopUpElement).isEmpty(), "Pop up about sent message doesn't shown");
58          return this;
59      }

61      /**
62       * @param folderName   -- name of folder in which is necessary to check email presence
63       * @param emailSubject -- subject of email which need to check
64       * @return UkrNetHomePage;
65       */
66  
67      @Step("Checking presence email by subject: {1} in folder")
68      public UkrNetHomePage checkingPresenceEmailInFolder(WebElement folderName, String emailSubject, String message) {
69          click(folderName);
70          WebElement titleBy = getDynamicElement(emailSubject, parentElement);
71          assertTrue(titleBy, readText(titleBy).contains(emailSubject), message);
72          return this;
73      }

87      @Step("Download file")
88      public UkrNetHomePage downloadFile(String emailSubject, String fileLink) {
89          click(inboxFolder);
90          openEmail(emailSubject);
91          try {
92              driver.get(fileLink);
93          } catch (org.openqa.selenium.TimeoutException e){
94              return this;
95          }
96          return this;
97      }

99      @Step("Check file downloading")
100     public void checkFileDownloading(String fileDirectory, String fileName) throws InterruptedException {
101         Thread.sleep(1000);
102         File file = new File(fileDirectory);
103         File[] dir_contents = file.listFiles();
104         String nameAndDirectory = fileDirectory + "/" + fileName;
105         File finalPath = new File(nameAndDirectory);
106         boolean isFile =finalPath.exists();
107         for(int i =0; i < dir_contents.length; i++){
108             if(dir_contents[i].getName().equals(fileName)){
109                 isFile = true;
110             }
111         }
112         if (isFile) {
113             finalPath.delete();
114         }
115         assertTrue(isFile, "File was not downloaded");
116     }
