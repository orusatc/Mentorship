3   import io.github.bonigarcia.wdm.WebDriverManager;
4   import org.openqa.selenium.WebDriver;
5   import org.openqa.selenium.firefox.FirefoxDriver;
6   import org.openqa.selenium.firefox.FirefoxProfile;
7   import org.openqa.selenium.remote.CapabilityType;
8   import org.openqa.selenium.remote.DesiredCapabilities;
9   
10  public class MozillaDriverManager extends DriverManager {
11      DesiredCapabilities capabilities;
12      private String downloadFilepathMozilla = "F:\\IT\\Mentorship\\src\\test\\java\\downloadsMozilla";
13  
14  
15      @Override
16      protected WebDriver createDriver() {
17          FirefoxProfile profile = new FirefoxProfile();
18          profile.setPreference("browser.download.folderList", 2);
19          profile.setPreference("browser.download.manager.showWhenStarting", false);
20          profile.setPreference("browser.download.dir", downloadFilepathMozilla);
21          profile.setPreference("browser.helperApps.neverAsk.openFile", "image/jpeg");
22          profile.setPreference("browser.helperApps.neverAsk.saveToDisk", "image/jpeg");
23  
24          profile.setPreference("browser.helperApps.alwaysAsk.force", false);
25          profile.setPreference("browser.download.manager.useWindow", false);
26          profile.setPreference("browser.download.manager.focusWhenStarting", false);
27          profile.setPreference("browser.download.manager.alertOnEXEOpen", false);
28          profile.setPreference("browser.download.manager.showAlertOnComplete", false);
29          profile.setPreference("browser.download.manager.closeWhenDone", true);
30  
31          capabilities = DesiredCapabilities.firefox();
32          capabilities.setCapability(FirefoxDriver.PROFILE, profile);
33          capabilities.setCapability(CapabilityType.ELEMENT_SCROLL_BEHAVIOR, 1);
34          //capabilities.setCapability("marionette", true);
35  
36          WebDriverManager.firefoxdriver().setup();
37          driver = new FirefoxDriver(capabilities);
38  
39          //Maximize Window
40          driver.manage().window().maximize();
41          DriverManager.createdDrivers.add(driver);
42          return driver;
43      }
44  }
