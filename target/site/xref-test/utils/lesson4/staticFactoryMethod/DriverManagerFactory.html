3   import org.openqa.selenium.WebDriver;

5   public class DriverManagerFactory {
6       static WebDriver driver;
7       public static WebDriver getManager(DriverType type) {
8           switch (type) {
9               case CHROME:
10                  //return already driver but not driver manager
11                  driver = new ChromeDriverManager().createDriver();
12                  break;
13              case FIREFOX:
14                  driver = new MozillaDriverManager().createDriver();
15                  break;
16              default:
17                  driver = new ChromeDriverManager().createDriver();
18                  break;
19          }
20          return driver;
21      }
22  
23      public static void quitDriver() {
24          for (WebDriver drivers : DriverManager.createdDrivers) {
25              drivers.quit();
26          }
27      }
28  
29      public static WebDriver getDriver() {
30          if(null == driver){
31              getManager(DriverType.CHROME);
32          }
33          return driver;
34      }
