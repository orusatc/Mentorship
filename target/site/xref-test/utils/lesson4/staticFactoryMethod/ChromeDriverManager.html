3   import io.github.bonigarcia.wdm.WebDriverManager;
4   import org.openqa.selenium.WebDriver;
5   import org.openqa.selenium.chrome.ChromeDriver;
6   import org.openqa.selenium.chrome.ChromeOptions;
7   import org.openqa.selenium.remote.CapabilityType;
8   import org.openqa.selenium.remote.DesiredCapabilities;
9   
10  import java.util.HashMap;
11  
12  public class ChromeDriverManager extends DriverManager {
13      DesiredCapabilities capabilities;
14      private String downloadFilepathChrome = "F:\\IT\\Mentorship\\src\\test\\java\\downloadsChrome";
15  
16      @Override
17      protected WebDriver createDriver() {
18          HashMap<String, Object> prefs = new HashMap<String, Object>();
19          prefs.put("profile.default_content_settings.popups", 0);
20          prefs.put("download.default_directory", downloadFilepathChrome);
21          ChromeOptions options = new ChromeOptions();
22          options.setExperimentalOption("prefs", prefs);
23          capabilities = DesiredCapabilities.chrome();
24          capabilities.setCapability(CapabilityType.ACCEPT_SSL_CERTS, true);
25          capabilities.setCapability(ChromeOptions.CAPABILITY, options);
26  
27          WebDriverManager.chromedriver().setup();
28          driver = new ChromeDriver(capabilities);
29          //Maximize Window
30          driver.manage().window().maximize();
31          DriverManager.createdDrivers.add(driver);
32          return driver;
33      }
34  }
